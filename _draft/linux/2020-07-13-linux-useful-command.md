# Linux批量清空当前目录中的日志文件
```
for i in `find . -name "*.log"`; do cat /dev/null >$i; done
for i in `find . -name "*.log"`;do >$i; done
for i in `find . -name "*.log" -o -name "*.out" -o -name "*.sql"`;do >$i; done
```
